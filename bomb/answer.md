int input[6];
read_six_numbers(input);

// validate
for (int i = 0; i < 6; i++) {
    if (input[i] < 1 || input[i] > 6) explode_bomb();
    for (int j = i+1; j < 6; j++)
        if (input[i] == input[j]) explode_bomb();
}

// transform
for (int i = 0; i < 6; i++) {
    input[i] = 7 - input[i];
}

// reorder linked list
Node *node_ptrs[6];
for (int i = 0; i < 6; i++) {
    node_ptrs[i] = get_node_by_index(input[i]);
}
for (int i = 0; i < 5; i++) {
    node_ptrs[i]->next = node_ptrs[i+1];
}
node_ptrs[5]->next = NULL;

// validate descending order
Node *p = node_ptrs[0];
for (int i = 0; i < 5; i++) {
    if (p->value < p->next->value) explode_bomb();
